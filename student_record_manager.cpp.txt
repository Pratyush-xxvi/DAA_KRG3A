// student_record_manager.cpp
#include <bits/stdc++.h>
using namespace std;

const int TABLE_SIZE = 10; // hash table buckets

// ==========================
// STUDENT CLASS
// ==========================
class Student {
public:
    int roll;
    string name;
    int age;
    string branch;

    Student() {}
    Student(int r, string n, int a, string b)
        : roll(r), name(n), age(a), branch(b) {}
};

// ==========================
// NODE for linked list in hash table
// ==========================
struct Node {
    Student data;
    Node* next;
    Node(Student s) : data(s), next(nullptr) {}
};

// ==========================
// HASH TABLE CLASS
// ==========================
class HashTable {
private:
    vector<Node*> table;

    // hash function
    int hashFunction(int key) {
        return key % TABLE_SIZE;
    }

public:
    HashTable() {
        table.resize(TABLE_SIZE, nullptr);
    }

    // Insert a student record
    void insert(Student s) {
        int index = hashFunction(s.roll);
        Node* newNode = new Node(s);

        // check duplicate roll
        Node* temp = table[index];
        while (temp) {
            if (temp->data.roll == s.roll) {
                cout << "âŒ Roll number already exists!\n";
                delete newNode;
                return;
            }
            temp = temp->next;
        }

        newNode->next = table[index];
        table[index] = newNode;
        cout << "âœ… Student added successfully!\n";
    }

    // Search by roll number
    Student* search(int roll) {
        int index = hashFunction(roll);
        Node* temp = table[index];
        while (temp) {
            if (temp->data.roll == roll)
                return &(temp->data);
            temp = temp->next;
        }
        return nullptr;
    }

    // Delete a record
    bool remove(int roll) {
        int index = hashFunction(roll);
        Node* temp = table[index];
        Node* prev = nullptr;

        while (temp) {
            if (temp->data.roll == roll) {
                if (prev) prev->next = temp->next;
                else table[index] = temp->next;
                delete temp;
                return true;
            }
            prev = temp;
            temp = temp->next;
        }
        return false;
    }

    // Display all records
    void displayAll() {
        cout << "\n========= STUDENT RECORDS =========\n";
        for (int i = 0; i < TABLE_SIZE; i++) {
            Node* temp = table[i];
            if (temp) cout << "Bucket[" << i << "]: ";
            while (temp) {
                cout << "[" << temp->data.roll << ", "
                     << temp->data.name << ", "
                     << temp->data.age << ", "
                     << temp->data.branch << "] -> ";
                temp = temp->next;
            }
            if (table[i]) cout << "NULL\n";
        }
        cout << "==================================\n";
    }

    // Save all records to file
    void saveToFile(string filename) {
        ofstream fout(filename);
        for (int i = 0; i < TABLE_SIZE; i++) {
            Node* temp = table[i];
            while (temp) {
                fout << temp->data.roll << " "
                     << temp->data.name << " "
                     << temp->data.age << " "
                     << temp->data.branch << "\n";
                temp = temp->next;
            }
        }
        fout.close();
        cout << "ðŸ’¾ Data saved to file successfully!\n";
    }

    // Load records from file
    void loadFromFile(string filename) {
        ifstream fin(filename);
        if (!fin) {
            cout << "âš ï¸ No existing data file found.\n";
            return;
        }
        int r, a;
        string n, b;
        while (fin >> r >> n >> a >> b) {
            insert(Student(r, n, a, b));
        }
        fin.close();
        cout << "ðŸ“‚ Data loaded from file successfully!\n";
    }
};

// ==========================
// MAIN FUNCTION
// ==========================
int main() {
    HashTable ht;
    ht.loadFromFile("students.txt");

    int choice;
    while (true) {
        cout << "\n======= STUDENT RECORD MANAGER =======\n";
        cout << "1. Add Student\n";
        cout << "2. Search Student\n";
        cout << "3. Delete Student\n";
        cout << "4. Display All\n";
        cout << "5. Save & Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        if (choice == 1) {
            int roll, age;
            string name, branch;
            cout << "Enter Roll: "; cin >> roll;
            cout << "Enter Name: "; cin >> name;
            cout << "Enter Age: "; cin >> age;
            cout << "Enter Branch: "; cin >> branch;
            ht.insert(Student(roll, name, age, branch));
        }
        else if (choice == 2) {
            int roll;
            cout << "Enter Roll No to search: ";
            cin >> roll;
            Student* s = ht.search(roll);
            if (s)
                cout << "âœ… Found: " << s->roll << " "
                     << s->name << " " << s->age << " " << s->branch << "\n";
            else
                cout << "âŒ Record not found!\n";
        }
        else if (choice == 3) {
            int roll;
            cout << "Enter Roll No to delete: ";
            cin >> roll;
            if (ht.remove(roll))
                cout << "ðŸ—‘ï¸ Record deleted successfully!\n";
            else
                cout << "âŒ Record not found!\n";
        }
        else if (choice == 4) {
            ht.displayAll();
        }
        else if (choice == 5) {
            ht.saveToFile("students.txt");
            cout << "ðŸ‘‹ Exiting...\n";
            break;
        }
        else {
            cout << "âš ï¸ Invalid choice!\n";
        }
    }
    return 0;
}
